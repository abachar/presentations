Constants


'use strict';

const PI = 3.14;
console.log(PI); // Will print 3.14

PI = 10;
// => TypeError: Assignment to constant variable.

Scoping


function scoping() {
	var withVar = 10;
	let withLet = 20;

	if (true) {
		var withVar = 11;
		let withLet = 21;

		console.log(withVar);  // 11
		console.log(withLet);  // 21
	}
	console.log(withVar);  // 11
	console.log(withLet);  // 20
}

Arrow function


let numbers = [1, 4, 9];
let doubles = numbers.map(n => num * 2);

	return Math.sqrt(n);
});

Function parameters


function defaultParameters(x, y = 7, z = 42) {
	return x + y + z
}
defaultParameters(1) === 50

function restParameter(x, y, ...a) {
	return (x + y) * a.length
}
restParameter(1, 2, "hello", true, 7) === 9

String interpolation


var x = 1;
var y = 2;
`${ x } + ${ y } = ${ x + y}`; // "1 + 2 = 3"

Destructuring


var list = [1, 2, 3];
var [a, , b] = list;
[b, a] = [a, b];

var today = { d: 6, m: 2, y: 2013 };
var { m: month, y: year } = today; // month = 2, year = 2013

function paramDestructuring([name, val]) {
	console.log(name, val)
}
paramDestructuring([ "bar", 42 ]);

Classes


class Shape {
	constructor (x, y) {
		this.move(x, y)
	}
	move (x, y) {
		this.x = x
		this.y = y
	}
}

Classes


class Circle extends Shape {
	constructor(x, y, radius) {
		super(x, y)
		this.radius = radius
	}

	set radius(radius)  { this.radius = radius; }
	get radius()        { return this.radius;   }
}

var circle = new Circle(0, 0 50);
circle.radius === 50;

For of/in


let arr = [3, 5, 7];
arr.foo = "hello";

for (let i in arr) {
	console.log(i); // logs "0", "1", "2", "foo"
}

for (let i of arr) {
	console.log(i); // logs "3", "5", "7"
}

s.size === 2
s.has("hello") === true
for (let val of s.values()) // insertion order
	console.log(val)
